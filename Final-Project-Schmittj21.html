EU_countries <- read.csv("eucountries.csv", header = TRUE)
countries_ALL <- dbSendQuery(db, 'SELECT
                              customers.Country,
                              customers.Customer_ID,
                              invoices.invoice_date,
                              invoice_products.invoice_no as "invoice_nbr",
                              invoice_products.SKU,
                              products.description,
                              invoice_products.quantity,
                              invoice_products.actual_unit_price
        FROM Customers
        JOIN invoices USING (customer_id)
        JOIN invoice_products USING (invoice_no)
        JOIN products USING (SKU)
        WHERE country != "United Kingdom"')

OLR_EU_Product_Sales <- fetch(countries_ALL, n=-1)
